112 ! IF(SM(I,J)/=SPVAL .AND. THZ0(I,J)/=SPVAL .AND. &
113 ! THS(I,J)/=SPVAL)THEN
114 ! TSFC = (SM(I,J)*THZ0(I,J)+(1.-SM(I,J))*THS(I,J)) &
115 ! *(PINT(I,J,NINT(LMH(I,J))+1)/P1000)**CAPA
116 ! Per AWC's request, use 2m T instead of skin T so that freezing level

127 ! IF(SM(I,J)/=SPVAL .AND. QZ0(I,J)/=SPVAL .AND. &
128 ! QS(I,J)/=SPVAL)THEN
129 ! QSFC = SM(I,J)*QZ0(I,J)+(1.-SM(I,J))*QS(I,J)

138  END IF

141  es=fpvsnew(tsfc)

175  pfrz = exp(alpfrz)
176  pfrzl(i,j) = pfrz
177  IF(modelname == 'GFS' .OR.modelname == 'RAPR')THEN
178  es=fpvsnew(tfrz)
179  es=min(es,pfrz)
180  qsfrz=con_eps*es/(pfrz+con_epsm1*es)
181  ELSE
182  qsfrz=pq0/pfrz &
183  *exp(a2*(tfrz-a3)/(tfrz-a4))
184  END IF
185 !
186  rhz = qfrz/qsfrz
187  rhz = amax1(0.01,rhz)

189  rhfrz(i,j) = rhz
190 !
191  ELSE

193  zl = htsfc+2.0
194  delz = zu-zl
195  IF(tshltr(i,j)/=spval .AND. pshltr(i,j)/=spval)THEN
196  tsfc=tshltr(i,j)*(pshltr(i,j)*1.e-5)**capa
197  ELSE
198 ! GFS analysis does not have flux file to retrieve TSFC from
199  tsfc=t(i,j,lm)+d0065*(zmid(i,j,lm)-htsfc-2.0)
200  END IF
201  delt = t(i,j,l)-tsfc
202  zfrz(i,j) = zl + (tfrz-tsfc)/delt*delz
203 !

205 ! GFS does not output QS
206  IF(qshltr(i,j)/=spval)THEN
207  qsfc=qshltr(i,j)
208  ELSE
209  qsfc=q(i,j,lm)
210  END IF
211  delq = q(i,j,l)-qsfc
212  qfrz = qsfc + delq/delz*dzabv
213  qfrz = amax1(0.0,qfrz)
214 !
215  alph = alog(pmid(i,j,l))
216  alpl = alog(psfc)

218  alpfrz = alpl + delalp/delz*dzabv
219  pfrz = exp(alpfrz)
