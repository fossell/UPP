279 subroutine rsearch1(km1,z1,km2,z2,l2)
280 !$$$ subprogram documentation block
281 !
282 ! subprogram: rsearch1 search for a surrounding real interval
283 ! prgmmr: iredell org: w/nmc23 date: 98-05-01
284 !
285 ! abstract: this subprogram searches a monotonic sequences of real numbers
286 ! for intervals that surround a given search set of real numbers.
287 ! the sequences may be monotonic in either direction; the real numbers
288 ! may be single or double precision.
289 !
290 ! program history log:
291 ! 1999-01-05 mark iredell
292 !
293 ! usage: call rsearch1(km1,z1,km2,z2,l2)
294 ! input argument list:
295 ! km1 integer number of points in the sequence
296 ! z1 real (km1) sequence values to search
297 ! (z1 must be monotonic in either direction)
298 ! km2 integer number of points to search for
299 ! z2 real (km2) set of values to search for
300 ! (z2 need not be monotonic)
301 !
302 ! output argument list:
303 ! l2 integer (km2) interval locations from 0 to km1
304 ! (z2 will be between z1(l2) and z1(l2+1))
305 !
306 ! subprograms called:
307 ! sbsrch essl binary search
308 ! dbsrch essl binary search
309 !
310 ! remarks:
311 ! returned values of 0 or km1 indicate that the given search value
312 ! is outside the range of the sequence.
313 !
314 ! if a search value is identical to one of the sequence values
315 ! then the location returned points to the identical value.
316 ! if the sequence is not strictly monotonic and a search value is
317 ! identical to more than one of the sequence values, then the
318 ! location returned may point to any of the identical values.
319 !
320 ! if l2(k)=0, then z2(k) is less than the start point z1(1)
321 ! for ascending sequences (or greater than for descending sequences).
322 ! if l2(k)=km1, then z2(k) is greater than or equal to the end point
323 ! z1(km1) for ascending sequences (or less than or equal to for
324 ! descending sequences). otherwise z2(k) is between the values
325 ! z1(l2(k)) and z1(l2(k+1)) and may equal the former.
326 !
327 ! attributes:
328 ! language: fortran
329 !
330 !$$$
331  implicit none
332  integer,intent(in):: km1,km2
333  real,intent(in):: z1(km1),z2(km2)
334  integer,intent(out):: l2(km2)
335  integer k1,k2
336 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
337 ! find the surrounding input interval for each output point.
338  if(z1(1) <= z1(km1)) then
339 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
340 ! input coordinate is monotonically ascending.
341  do k2=1,km2
342  if (z1(1) >= z2(k2)) then
343  l2(k2) = 1
344  else
345  l2(k2)=km1
346  do k1=1,km1-1
347  if(z1(k1) <= z2(k2) .and. z1(k1+1) > z2(k2)) then
348  l2(k2) = k1
349  exit
350  endif
351  enddo
352  endif
353  enddo
354  else
355 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
356 ! input coordinate is monotonically descending.
357  do k2=1,km2
358  if (z1(1) <= z2(k2)) then
359  l2(k2) = 1
360  else
361  l2(k2)=km1
362  do k1=km1,2,-1
363  if(z2(k2) >= z1(k1) .and. z2(k2) < z1(k1-1)) then
364  l2(k2) = k1-1
365  exit
366  endif
367  enddo
368  endif
369  enddo
370  endif
371 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
372 end subroutine

374  subroutine tpause(km,p,u,v,t,h,ptp,utp,vtp,ttp,htp,shrtp)
375 !$$$ Subprogram documentation block
376 !
377 ! Subprogram: tpause Compute tropopause level fields
378 ! Prgmmr: Iredell Org: np23 Date: 1999-10-18
379 !
380 ! Abstract: This subprogram finds the tropopause level and computes fields
381 ! at the tropopause level. The tropopause is defined as the lowest level
382 ! above 500 mb which has a temperature lapse rate of less than 2 K/km.
383 ! The lapse rate must average less than 2 K/km over a 2 km depth.
384 ! If no such level is found below 50 mb, the tropopause is set to 50 mb.
385 ! The tropopause fields are interpolated linearly in lapse rate.
386 ! The tropopause pressure is found hydrostatically.
387 ! The tropopause wind shear is computed as the partial derivative
388 ! of wind speed with respect to height at the tropopause level.
389 !
390 ! Program history log:
391 ! 1999-10-18 Mark Iredell
392 !
393 ! Usage: call tpause(km,p,u,v,t,h,ptp,utp,vtp,ttp,htp,shrtp)
394 ! Input argument list:
395 ! km integer number of levels
396 ! p real (km) pressure (Pa)
397 ! u real (km) x-component wind (m/s)
398 ! v real (km) y-component wind (m/s)
399 ! t real (km) temperature (K)
400 ! h real (km) height (m)
401 ! Output argument list:
402 ! ptp real tropopause pressure (Pa)
403 ! utp real tropopause x-component wind (m/s)
404 ! vtp real tropopause y-component wind (m/s)
405 ! ttp real tropopause temperature (K)
406 ! htp real tropopause height (m)
407 ! shrtp real tropopause wind shear (1/s)
408 !
409 ! Files included:
410 ! physcons.h Physical constants
411 !
412 ! Subprograms called:
413 ! rsearch1 search for a surrounding real interval
414 !
415 ! Attributes:
416 ! Language: Fortran 90
417 !
418 !$$$
419  use physcons_post, only: con_rog
420  implicit none
421  integer,intent(in):: km
422  real,intent(in),dimension(km):: p,u,v,t,h
423  real,intent(out):: ptp,utp,vtp,ttp,htp,shrtp
424  real,parameter:: ptplim(2)=(/500.e+2,50.e+2/),gamtp=2.e-3,hd=2.e+3
425  real gamu,gamd,td,gami,wtp,spdu,spdd
426  integer klim(2),k,kd,ktp
427 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
428 ! find tropopause level
429  call rsearch1(km-2,p(2),2,ptplim,klim)
430  klim(1)=klim(1)+1
431  klim(2)=klim(2)+2
432  ! klim(1) > klim(2) or loops does not run ; klim(2) has a
433  ! minimum value of 3 to insure k-2 != 0 in called subprogram
434  gamd=1.e+9
435  ktp=klim(2)
436  wtp=0
437 ! do k=klim(1),klim(2)
438  do k=klim(1),klim(2),-1
439 ! gamu=(t(k-1)-t(k+1))/(h(k+1)-h(k-1))
440  gamu=(t(k+1)-t(k-1))/(h(k-1)-h(k+1))
441  if(gamu<=gamtp) then
442 ! call rsearch1(km-k-1,h(k+1),1,h(k)+hd,kd)
443  call rsearch1(k-2,h(2),1,h(k)+hd,kd)
444 ! td=t(k+kd)+(h(k)+hd-h(k+kd))/(h(k+kd+1)-h(k+kd))*(t(k+kd+1)-t(k+kd))
445  td=t(kd+2)+(h(k)+hd-h(2+kd))/(h(kd+1)-h(2+kd))*(t(kd+1)-t(2+kd))
446  gami=(t(k)-td)/hd
447  if(gami<=gamtp) then
448  ktp=k
449  wtp=(gamtp-gamu)/(max(gamd,gamtp+0.1e-3)-gamu)
450  exit
451  endif
452  endif
453  gamd=gamu
454  enddo
455 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
456 ! compute tropopause level fields
457  utp=u(ktp)-wtp*(u(ktp)-u(ktp-1))
458  vtp=v(ktp)-wtp*(v(ktp)-v(ktp-1))
459  ttp=t(ktp)-wtp*(t(ktp)-t(ktp-1))
460  htp=h(ktp)-wtp*(h(ktp)-h(ktp-1))
461  ptp=p(ktp)*exp((h(ktp)-htp)*(1-0.5*(ttp/t(ktp)-1))/(con_rog*t(ktp)))
462  spdu=sqrt(u(ktp)**2+v(ktp)**2)
463  spdd=sqrt(u(ktp-1)**2+v(ktp-1)**2)
464  shrtp=(spdu-spdd)/(h(ktp)-h(ktp-1))
465 
466  utp=u(ktp)-wtp*(u(ktp)-u(ktp+1))
467  vtp=v(ktp)-wtp*(v(ktp)-v(ktp+1))
468  ttp=t(ktp)-wtp*(t(ktp)-t(ktp+1))
469  htp=h(ktp)-wtp*(h(ktp)-h(ktp+1))
470  ptp=p(ktp)*exp((h(ktp)-htp)*(1-0.5*(ttp/t(ktp)-1))/(con_rog*t(ktp)))
471  spdu=sqrt(u(ktp)**2+v(ktp)**2)
472  spdd=sqrt(u(ktp+1)**2+v(ktp+1)**2)
473  shrtp=(spdu-spdd)/(h(ktp)-h(ktp+1))
474  end subroutine

580 subroutine mptgen(mpirank,mpisize,nd,jt1,jt2,j1,j2,jx,jm,jn)
581 !$$$ Subprogram documentation block
582 !
583 ! Subprogram: mptgen Generate grid decomposition dimensions
584 ! Prgmmr: Iredell Org: W/NP23 Date: 1999-02-12
585 !
586 ! Abstract: This subprogram decomposes total dimensions of a problem
587 ! into smaller domains to be managed on a distributed memory system.
588 ! The last dimension given is decomposed first. If more decompositions
589 ! are possible, the next to last dimension is decomposed next, and so on.
590 ! The transpositions between decompositions should be done by mptran*.
591 !
592 ! Program history log:
593 ! 1999-02-12 Mark Iredell

596 ! Input argument list:
597 ! mpirank integer(kint_mpi) rank of the process (from mpi_comm_rank)
598 ! mpisize integer(kint_mpi) size of the process (from mpi_comm_size)
599 ! nd integer(kint_mpi) number of dimensions to decompose
600 ! jt1 integer(kint_mpi) (nd) lower bounds of total dimensions
601 ! jt2 integer(kint_mpi) (nd) upper bounds of total dimensions
602 ! Output argument list:
603 ! j1 integer(kint_mpi) (nd) lower bounds of local decompositions
604 ! j2 integer(kint_mpi) (nd) upper bounds of local decompositions
605 ! jx integer(kint_mpi) (nd) local size of decompositions
606 ! jm integer(kint_mpi) (nd) maximum size of decompositions
607 ! jn integer(kint_mpi) (nd) number of decompositions
608 !
609 ! Attributes:
610 ! Language: Fortran 90
611 !
612 !$$$
613  use machine_post,only:kint_mpi
614  implicit none
615  integer(kint_mpi),intent(in):: mpirank,mpisize,nd,jt1(nd),jt2(nd)
616  integer(kint_mpi),intent(out):: j1(nd),j2(nd),jx(nd),jm(nd),jn(nd)
617  integer msize,mrank,msn,mrn,n
618 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
619  msize=mpisize
620  mrank=mpirank
621  do n=nd,1,-1
622  if(jt2(n)>=jt1(n)) then
623  jm(n)=(jt2(n)-jt1(n))/msize+1
624  msn=max(msize/(jt2(n)-jt1(n)+1),1)
625  if(n==1) msn=1
626  jn(n)=msize/msn
627  mrn=mrank/msn
628  j1(n)=min(jt1(n)+jm(n)*mrn,jt2(n)+1)
629  j2(n)=min(jt1(n)+jm(n)*mrn+jm(n)-1,jt2(n))
630  jx(n)=j2(n)-j1(n)+1
631  msize=msn
632  mrank=mod(mrank,msn)
633  write(0,*)' mrank=',mrank,' j1=',j1(n),' j2=',j2(n),' jx=',jx(n),' jm=',jm(n)
634  else
635  jm(n)=0
636  jn(n)=1
637  j1(n)=jt1(n)
638  j2(n)=jt2(n)
639  jx(n)=0
640  endif
641  enddo
642 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
643 end subroutine

648 !
649 ! Subprogram: mptranr4 Transpose grid decompositions
650 ! Prgmmr: Iredell Org: W/NP23 Date: 1999-02-12
651 !
652 ! Abstract: This subprogram transposes an array of data from one
653 ! grid decomposition to another by using message passing.
654 ! The grid decompositions should be generated by mptgen.
655 !
656 ! Program history log:
657 ! 1999-02-12 Mark Iredell
658 !
659 ! Usage: call mptranr4(mpicomm,mpisize,im,ida,idb,&
660 ! jm,jma,jmb,jda,km,kma,kmb,kdb,a,b)
661 ! Input argument list:
662 ! mpicomm integer(kint_mpi) mpi communicator
663 ! mpisize integer(kint_mpi) size of the process (from mpi_comm_size)
664 ! im integer(kint_mpi) undecomposed range
665 ! ida integer(kint_mpi) undecomposed input dimension
666 ! idb integer(kint_mpi) undecomposed output dimension
667 ! jm integer(kint_mpi) output grid decomposition size
668 ! jma integer(kint_mpi) input grid undecomposed range
669 ! jmb integer(kint_mpi) output grid decomposed range
670 ! jda integer(kint_mpi) input grid undecomposed dimension
671 ! km integer(kint_mpi) input grid decomposition size
672 ! kma integer(kint_mpi) input grid decomposed range
673 ! kmb integer(kint_mpi) output grid undecomposed range
674 ! kdb integer(kint_mpi) output grid undecomposed dimension
675 ! a real(4) (ida,jda,kma) input array
676 ! Output argument list:
677 ! b real(4) (idb,kdb,jmb) output array
678 ! ta,tb real(4) (im,jm,km,mpisize) work arrays
679 !
680 ! Subprograms called:
681 ! mpi_alltoall mpi exchange of data between every process pair
682 !
683 ! Remarks:
684 ! While this routine serves a wide variety of scalable transpose functions
685 ! for multidimensional grids,
686 ! (a) it does not work with nonrectanguloid grids;
687 ! (b) it does not do any load balancing;
688 ! (c) it does not do any communication hiding.
689 !
690 ! This subprogram must be used rather than mpi_alltoall
691 ! in any of the following cases:
692 ! (a) The undecomposed range is less than the respective dimension
693 ! (either im<ida or im<idb)
694 ! (b) The decomposition size is greater than one
695 ! (either km>1 or jm>1)
696 ! (c) The decomposed range is ever zero
697 ! (either kma==0 or jmb==0 for any process)
698 ! (d) The output grid range is not the full extent
699 ! (either kmb<mpisize or kmb<kda or jma<mpisize or jma<jda)

701 ! rather than this subprogram and would be more efficient.
702 !
703 ! Example 1. Transpose a 1000 x 10000 matrix.
704 !
705 !!! include 'mpif.h' ! use mpi
706 !!! parameter(jt=1000,kt=10000) ! set problem size
707 !!! real,allocatable:: a(:,:),b(:,:) ! declare arrays
708 !!! call mpi_init(ierr) ! initialize mpi
709 !!! call mpi_comm_rank(MPI_COMM_WORLD,mpirank,ierr) ! get mpi rank
710 !!! call mpi_comm_size(MPI_COMM_WORLD,mpisize,ierr) ! get mpi size
711 !!! call mptgen(mpirank,mpisize,1,1,jt,j1,j2,jx,jm,jn) ! decompose output
712 !!! call mptgen(mpirank,mpisize,1,1,kt,k1,k2,kx,km,kn) ! decompose input
713 !!! allocate(a(jt,k1:k2),b(kt,j1:j2)) ! allocate arrays
714 !!! a=reshape((/((j+k,j=1,jt),k=k1,k2)/),(/jt,k2-k1+1/)) ! initialize input
715 !!! call mptranr4(MPI_COMM_WORLD,mpisize,1,1,1, ! transpose arrays
716 !!! & jm,jt,j2-j1+1,jt,km,k2-k1+1,kt,kt,a,b)
717 !!! print '(2i8,f16.1)',((k,j,b(k,j),k=2000,kt,2000), ! print some values
718 !!! & j=((j1-1)/200+1)*200,j2,200)

722 ! This transpose took 0.6 seconds on 4 2-way winterhawk nodes.
723 ! A 20000x10000 transpose took 3.4 seconds on 16 2-way winterhawk nodes.
724 ! Thus a transpose may take about 1 second for every 16 Mb per node.
725 !
726 ! Attributes:
727 ! language: fortran 90
728 !
729 !$$$
730  use machine_post,only:kint_mpi
731  implicit none
732  include 'mpif.h'
733  integer(kint_mpi),intent(in):: mpicomm,mpisize
734  integer(kint_mpi),intent(in):: im,ida,idb
735  integer(kint_mpi),intent(in):: jm,jma,jmb,jda
736  integer(kint_mpi),intent(in):: km,kma,kmb,kdb
737  real(4),dimension(ida,jda,kma),intent(in):: a
738  real(4),dimension(idb,kdb,jmb),intent(out):: b
739  real(4),dimension(im,jm,km,mpisize),intent(inout):: ta,tb
740  integer(4) jmb1(1),jmbf(mpisize),kma1(1),kmaf(mpisize)
741  integer(kint_mpi)::i,j,k,l,ierr,ja,kb
742 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
743  jmb1(1) = jmb
744  call mpi_allgather(jmb1,1,mpi_integer4,jmbf,1,mpi_integer4,mpicomm,ierr)
745  kma1(1) = kma
746  call mpi_allgather(kma1,1,mpi_integer4,kmaf,1,mpi_integer4,mpicomm,ierr)

748 ! internally transpose input array
749 !$omp parallel do private(i,j,k,l,ja)
750  do l=1,mpisize
751  do k=1,kma
752  do j=1,jm
753  ja = j + sum(jmbf(1:l-1))
754  if(ja <= jma) then
755  do i=1,im
756  ta(i,j,k,l) = a(i,ja,k)
757  enddo
758  endif
759  enddo
760  enddo
761  enddo
762 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
763 ! externally transpose data
764  call mpi_alltoall(ta,im*jm*km,mpi_real4,tb,im*jm*km,mpi_real4,mpicomm,ierr)

766 ! internally transpose output array
767 !$omp parallel do private(i,j,k,l,kb)
768  do l=1,mpisize
769  do k=1,km
770  kb = k + sum(kmaf(1:l-1))
771  if(kb <= kmb) then
772  do j=1,jmb
773  do i=1,im
774  b(i,kb,j) = tb(i,j,k,l)
775  enddo
776  enddo
777  endif
778  enddo
779  enddo

781 end subroutine
